def extract_features(image_path):
    # Load the image
    image = cv2.imread(image_path)
    if image is None:
        raise ValueError("Image not found or unable to load.")

    # Convert to grayscale
    gray_image = color.rgb2gray(image)
    gray_image = img_as_ubyte(gray_image)  # Convert to 8-bit unsigned integer

    # Resize the image (optional, for consistency)
    resized_image = cv2.resize(gray_image, (256, 256))

    # Noise reduction using Gaussian Blur
    blurred_image = cv2.GaussianBlur(resized_image, (5, 5), 0)

    # Contrast enhancement using CLAHE
    clahe = cv2.createCLAHE(clipLimit=2.0, tileGridSize=(8, 8))
    enhanced_image = clahe.apply(blurred_image)

    # Feature 1: Color Histogram (for grayscale intensity distribution)
    histogram = cv2.calcHist([enhanced_image], [0], None, [256], [0, 256])
    histogram = histogram.flatten()

    # Feature 2: Texture (using Gray-Level Co-occurrence Matrix - GLCM)
    glcm = graycomatrix(enhanced_image, distances=[1], angles=[0], levels=256, symmetric=True, normed=True)
    contrast = graycoprops(glcm, 'contrast')[0, 0]
    energy = graycoprops(glcm, 'energy')[0, 0]

    # Feature 3: Shape and Size (using contours and region properties)
    _, binary_image = cv2.threshold(enhanced_image, 128, 255, cv2.THRESH_BINARY)
    contours, _ = cv2.findContours(binary_image, cv2.RETR_EXTERNAL, cv2.CHAIN_APPROX_SIMPLE)

    # Find the largest contour (assumed to be the pepper seed)
    largest_contour = max(contours, key=cv2.contourArea)
    area = cv2.contourArea(largest_contour)  # Area of the largest contour
    perimeter = cv2.arcLength(largest_contour, True)  # Perimeter of the largest contour

    # Feature 4: Aspect Ratio (width/height of bounding box)
    x, y, w, h = cv2.boundingRect(largest_contour)
    aspect_ratio = float(w) / h

    # Return extracted features
    features = {
        "histogram": histogram,
        "contrast": contrast,
        "energy": energy,
        "area": area,
        "perimeter": perimeter,
        "aspect_ratio": aspect_ratio
    }
    return features
